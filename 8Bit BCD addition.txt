; Addition of 8 bit operator

.model small
.stack 100H

.data
MSG1 DB 'Enter first 8 bit number: $'
MSG2 DB 0DH,0AH,'Enter second 8 bit number: $'
MSG3 DB 0DH,0AH,'Sum is: $'

.code
MAIN PROC
    MOV AX, @DATA
    MOV DS, AX

    LEA DX, MSG1
    MOV AH, 09H
    INT 21H

    MOV AH, 01H
    INT 21H
    SUB AL, 30H
    MOV BL, AL

    LEA DX, MSG2
    MOV AH, 09H
    INT 21H

    MOV AH, 01H
    INT 21H
    SUB AL, 30H
    ADD BL, AL

    LEA DX, MSG3
    MOV AH, 09H
    INT 21H

    MOV AL, BL
    ADD AL, 30H
    MOV DL, AL
    MOV AH, 02H
    INT 21H

    MOV AH, 4CH
    INT 21H
MAIN ENDP
END MAIN
