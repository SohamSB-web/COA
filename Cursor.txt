;Assembly program to implement cursor activities.

DATA SEGMENT
STR1 DB 0AH, 0DH, "1: UDERLINE CURSOR"
     DB 0AH, 0DH, "2: HALF BLOCK CURSOR"
     DB 0AH, 0DH, "3: FULL BLOCK CURSOR"
     DB 0AH, 0DH, "4: HIDE CURSOR"
     DB 0AH, 0DH, "OTHER KEY: EXIT"
     DB 0AH, 0DH, "$"

DATA ENDS

CODE SEGMENT
ASSUME CS:CODE, DS:DATA
START: MOV AX, DATA
       MOV DS, AX

       LEA DX, STR1
       MOV AH, 09H
       INT 21H

BACK:  MOV AH, 07H
       INT 21H

       MOV AH, 03H
       INT 10H
       MOV AH, 02H
       INT 10H

       CMP AL, '1'
       JE UDERLINE
       CMP AL, '2'
       JE HALFBLOCK
       CMP AL, '3'
       JE FULLBLOCK
       CMP AL, '4'
       JE HIDE
       JMP EXIT

UDERLINE: MOV CX, 0607h
          JMP NEXT
HALFBLOCK: MOV CX, 0407h

FULLBLOCK: MOV CX, 0007h
           JMP NEXT
HIDE: MOV CX, 0606h
      JMP NEXT

NEXT: MOV AH, 01H
      INT 10H
      JMP BACK

EXIT: MOV AH, 4CH
      INT 21H

CODE ENDS
END START
