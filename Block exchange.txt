.model small
.stack 100h
.data
arr1 db 5,6,7,8
arr2 db 1,2,3,4
LEN equ 4
.code
main proc

mov ax,@data
mov ds,ax   
mov es,ax

mov cx, LEN
mov si, offset arr1
mov di, offset arr2

exch_loop:
	mov al, [si]
	xchg al, [di]
	mov [si], al
	inc si
	inc di
	loop exch_loop
	
mov ah,4ch
int 21h


main endp
end main
